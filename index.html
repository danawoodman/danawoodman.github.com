/*

Find me around the web.

Outputs a list of HTML link tags to the various places 
around the web you can find me.

*/
package main

import "fmt"

// Site represents an HTML link tag.
type Site struct {
  Label string
  Url   string
}

// Implement the Stringer interface on Site.
func (s *Site) String() string {
  return fmt.Sprintf("<a href=\"%s\">%s</a>", s.Url, s.Label)
}

func main() {

  // Heading
  fmt.Println("<h2>Find Me</h2>")

  sites := []Site{
    Site{"Hacker News", "https://news.ycombinator.com/user?id=danaw"},
    Site{"Github", "https://github.com/danawoodman"},
    Site{"StackOverflow", "http://stackoverflow.com/users/529829/dana-woodman"}
  }

  for _, site := range sites {
    fmt.Println(&site)
  }
}
